697 }
698
699 if (s->buff_pos >= s->decomp_bytes) {
700 int n = compression_decompress_incremental(&s->ds, s->decomp_buff,DECOMP_BUFF_SIZE32);
701 if (n == 0) {
702 return 0;
703 }
704 s->decomp_bytes = n;
705 s->buff_pos = 0;
706 }
707 char c = s->decomp_buff[s->buff_pos];
708 s->buff_pos += 1;
709 return c;
710}
711
712char peek_compressed(tokenizer_char_stream str, unsigned int n) {
713 tokenizer_compressed_state *s = (tokenizer_compressed_state*)str.state;
714
715 tokenizer_compressed_state old;
716
717 memcpy(&old, s, sizeof(tokenizer_compressed_state));
718
719 char c = get_compressed(str);;
720 for (unsigned int i = 1; i <= n; i ++) {
721 c = get_compressed(str);
722 }
723
724 memcpy(str.state, &old, sizeof(tokenizer_compressed_state));
725 return c;
726}
727
728
729void drop_compressed(tokenizer_char_stream str, unsigned int n) {
730 for (unsigned int i = 0; i < n; i ++) {
731 get_compressed(str);
732 }
733}

735
736VALUE tokpar_parse_compressed(char *bytes) {
737
738 tokenizer_compressed_state ts;
739
740 ts.decomp_bytes = 0;
741 memset(ts.decomp_buff, 0, 32);
742 ts.buff_pos = 0;
743
744 compression_init_state(&ts.ds, bytes);
745
746 tokenizer_char_stream str;
747 str.state = &ts;
748 str.more = more_compressed;
749 str.get = get_compressed;
750 str.peek = peek_compressed;
751 str.drop = drop_compressed;
752
753 return parse_program(str);
754}
