47/* Bawden's qq-expand-list implementation
48(define (qq-expand-list x)
49 (cond ((tag-comma? x)
50 `(list ,(tag-data x)))
51 ((tag-comma-atsign? x)
52 (tag-data x))
53 ((tag-backquote? x)
54 (qq-expand-list
55 (qq-expand (tag-data x))))
